```python
from functools import reduce

# Compose two functions
def compose(f, g):
    return lambda x: f(g(x))           # f(g(x))

double = lambda x: x * 2
add_one = lambda x: x + 1

double_then_add = compose(add_one, double)  # add_one(double(x))
add_then_double = compose(double, add_one)  # double(add_one(x))

print(double_then_add(5))   # 11  → double(5)=10, add_one(10)=11
print(add_then_double(5))   # 12  → add_one(5)=6, double(6)=12


# Compose any number of functions
def compose_all(*funcs):
    """Apply functions right to left: compose_all(f, g, h)(x) = f(g(h(x)))"""
    return reduce(compose, funcs)

def pipe_all(*funcs):
    """Apply functions left to right: pipe_all(f, g, h)(x) = h(g(f(x)))"""
    return reduce(lambda f, g: lambda x: g(f(x)), funcs)


strip     = str.strip
lower     = str.lower
normalize = pipe_all(strip, lower)     # strip first, then lower

print(normalize("  Hello World  "))    # "hello world"


# Practical pipeline
import re

clean_text = pipe_all(
    str.strip,
    str.lower,
    lambda s: re.sub(r"[^\w\s]", "", s),    # Remove punctuation
    lambda s: re.sub(r"\s+", " ", s),       # Normalize whitespace
)

print(clean_text("  Hello,   World!!!  "))  # "hello world"
```
